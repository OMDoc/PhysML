\documentclass{owl2onto}
\begin{ontology}[id=foaf,baseURI=http://xmlns.com/foaf/0.1/]
\declaration[type=Class,name=Agent,cseq=agent]
  {An agent (eg. person, group, software or physical artifact).}
\declaration[type=Class,name=Document,cseq=doc]
  {An agent (eg. person, group, software or physical artifact).}
\declaration[type=NamedIndividual,name=Peter,cseq=peter]
  {An agent (eg. person, group, software or physical artifact).}
\axiom{$\DisjointClasses{\agent,\doc}$}{Agents are not Documents}
\axiom{$\SubClassOf\agent\doc$}{Agents are Documents}
\end{ontology}
