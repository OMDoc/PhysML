\begin{module}[id=thermostatics]
\importmodule[units]{units}
\symdef{red}{{\sf{red}}}

\begin{observable}[name=temparature]
  \begin{opdef}
    Measure with a thermometer
  \end{opdef}
  \begin{refinement}
Give a rule to iteratively improve the thermometer (exp. setup)
  \end{refinement}
  \symtype{dimensions}{$\temperature$}
\end{observable}

\begin{observable}[name=density]
  \begin{opdef}
    opdef
  \end{opdef}
  \begin{refinement}
    give an iterative rule to improve stepwise the exp. setup
  \end{refinement}
  \symtype{dimensions}{$\mass/\volume$}
\end{observable}

  \sysref{gas-cylinder}

  \begin{measurement}[id=m-213]

      \statevalue{pressure}{$332.49586\cdot{\Newton\over\centimeter^2}$}

    \begin{state}[of=mercury-theormometer]
      \statevalue{temparature}{$17.52\cdot\Kelvin$}
    \end{state}
  \end{measurement}
\end{experiment}
